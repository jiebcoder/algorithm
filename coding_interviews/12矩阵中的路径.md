## 方法一：回溯法，boolean[] flag 标记是否访问过

```java

public boolean hasPath(char[] matrix, int rows, int cols, char[] str) {
    boolean[] flag = new boolean[matrix.length];
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (backtracking(matrix, rows, cols, str, i, j, flag, 0)) {
                return true;
            }
        }
    }
    return false;
}

private boolean backtracking(char[] matrix, int rows, int cols, char[] str, int i, int j, boolean[] flag, int idx) {
    int tmp = i * cols + j;  //注意
    if (i < 0 || i >= rows || j < 0 || j >= cols || matrix[tmp] != str[idx] || flag[tmp] == true) {
        return false;
    }
    if (idx == str.length - 1) {
        return true;
    }
    flag[tmp] = true;
    if (backtracking(matrix, rows, cols, str, i + 1, j, flag, idx + 1) ||
        backtracking(matrix, rows, cols, str, i - 1, j, flag, idx + 1) ||
        backtracking(matrix, rows, cols, str, i, j + 1, flag, idx + 1) ||
        backtracking(matrix, rows, cols, str, i, j - 1, flag, idx + 1)) {
        return true;
    }
    flag[tmp] = false;
    return false;
}
```

## 方法二：回溯法，修改原字符数组

```java

public boolean hasPath(char[] matrix, int rows, int cols, char[] str) {
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (backtracking(matrix, rows, cols, str, i, j, 0)) {
                return true;
            }
        }
    }
    return false;
}

private boolean backtracking(char[] matrix, int rows, int cols, char[] str, int i, int j, int idx) {
    int tmp = i * cols + j;  //注意
    if (i < 0 || i >= rows || j < 0 || j >= cols || matrix[tmp] != str[idx]) {
        return false;
    }
    if (idx == str.length - 1) {
        return true;
    }
    matrix[tmp] = '#';
    if (backtracking(matrix, rows, cols, str, i + 1, j, idx + 1) ||
        backtracking(matrix, rows, cols, str, i - 1, j, idx + 1) ||
        backtracking(matrix, rows, cols, str, i, j + 1, idx + 1) ||
        backtracking(matrix, rows, cols, str, i, j - 1, idx + 1)) {
        return true;
    }
    matrix[tmp] = str[idx];
    return false;
}
```
